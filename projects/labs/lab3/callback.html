Callback/Effects Example

This is the the way without the callback function, watch me first!

This is the way using the callback function, look how much better I am!

    	
        	/* Example borrowed from http://www.w3schools.com/jquery/jquery_callback.asp */
            var jq = jQuery.noConflict();
            jq(document).ready(function() {
                /* Without using the callback function */
                jq("button#button1").click(function() {
                    jq("p#p1").hide(1500);
                    alert("The paragraph is now hidden");
                });
                
                /* With using the callback function */
                jq("button#button2").click(function() {
                    jq("p#p2").hide(1500,function(){
                      alert("The paragraph is now hidden");
                    });
                });
                
                
                /* Toggle the view of the code */
                jq("button#toggleJQCode").click(function() {
                    jq("pre").slideToggle();
                });
            });

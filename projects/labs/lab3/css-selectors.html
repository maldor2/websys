            var jq = jQuery.noConflict();
            jq(document).ready(function(){
                /* 
                 * Changes all elements with the 'href' attribute = '#'
                 * to have background-color = 'red' and color = 'white'
                 * when the cursor mouses over tge element.
                 *
                 */
                jq("[href='#']").click(function() {
                    jq(this).css("background-color","red");
                    jq(this).css("color", "white");
                });
                
                /* 
                 * When the p element with class is clicked,
                 * the CSS is changed to add the font-size property with value 
                 * 2.5em, and background-color set to yellow.
                 *
                 */
                jq("p.change").click(function(){
                    jq(this).css({"font-size":"2.5em","background-color":"yellow", "color":"black"});
                });
                
                
                /* 
                 * Resets all a elements to their original state.
                 *
                 */
                jq("button#clearElements").click(function() {
                    jq("a").css("background-color","white");
                    jq("a").css("color", "blue");
                    jq("a").css("font-variant", "normal");
                    jq("p.change").css("background-color","white");
                    jq("p.change").css("font-size", "1.0em");
                });
                
                /* 
                 * Sets all the a elements with href != '#' to font-variant = 'small-caps'
                 * when the button with id = 'smallcaps' is clicked.
                 *
                 */
                jq("button#smallcaps").click(function() {
                    jq("a[href !='#']").css("font-variant", "small-caps");
                });
                
                /* 
                 * Uses the jQuery slideToggle effect to slide up all pre
                 * elements if display = none, or slide down all pre elements
                 * if display != none, when the button with id = 'toggleJQCode' is clicked.
                 * 
                 */
                jq("button#toggleJQCode").click(function() {
                    jq("pre").slideToggle();
                });
            });

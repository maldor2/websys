            var jq = jQuery.noConflict();
            jq(document).ready(function(){
                jq("[href='#']").mouseover(function() {
                    jq(this).css("background-color","red");
                    jq(this).css("color", "white");
                });
                jq("[href='#']").mouseout(function() {
                    jq(this).css("background-color","black");
                    jq(this).css("color", "white");
                });
                
                jq("p.remove").click(function(){
                    jq(this).html('Buh Bye!');
                    jq(this).css({"font-size":"2.5em","background-color":"yellow"});
                    jq(this).fadeOut(5000);
                });
                
                jq("button#clearLinks").click(function() {
                    jq("a").css("background-color","white");
                    jq("a").css("color", "blue");
                    jq("a").css("font-variant", "normal");
                });
                
                jq("button#smallcaps").click(function() {
                    jq("a[href !='#']").css("font-variant", "small-caps");
                });
                
                jq("button#toggleJQCode").click(function() {
                    jq("pre").slideToggle();
                });
            });
